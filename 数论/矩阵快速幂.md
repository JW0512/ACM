## 矩阵快速幂

```cpp
struct node {
	int mat[15][15];	//定义矩阵 
}x,y; 
node mul(node x,node y){	//矩阵乘法 
	node ans;
	for(int i=0;i<len;i++){
		for(int j=0;j<len;j++){
			ans.mat [i][j]=0;
			for(int k=0;k<len;k++){
				int tmp=(x.mat[i][k] * y.mat[k][j]) % mod;
				ans.mat[i][j]=(ans.mat[i][j]+tmp) % mod;
			}
		}
	}
	return ans;
}
node matpow(node x,node y,int num){		//矩阵快速幂 
	while(num){
		if(num&1)
			y=mul(y,x);
		x=mul(x,x);
		num>>=1;
	}
	return y;
}
```
