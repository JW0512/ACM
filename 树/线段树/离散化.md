## 贴海报
在长度为L的宣传栏上贴海报，问最后有几张海报能被看到

输入N,L代表N张海报，宣传栏长度为L
下面N行每行输入ai和bi，代表每张海报的范围
**注意：离散化一定要去重**

```cpp
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
int N, L;
const int maxn = 1e5 + 5;
int l[maxn], r[maxn];
int num[2 * maxn];
int vis[2 * maxn];
int ans;
#define Lson(x) x<<1
#define Rson(x) x<<1|1
struct TREE
{
	int l, r;
	int color;//着色
	int lazy;//更新的值
	inline void update(int val)
	{
		color = val;
		lazy = val;
	}
}tree[(2 * maxn) << 2];

void push_up(int x)//更新父节点的颜色
{
	if (tree[Lson(x)].color != tree[Rson(x)].color)//左右子树颜色不一样则颜色标记为-1
		tree[x].color = -1;
	else
		tree[x].color = tree[Lson(x)].color;//否则就是子树的颜色
}

void push_down(int x)//更新子树颜色
{
	if (!tree[x].lazy)
		return;
	int lazy = tree[x].lazy;
	tree[Lson(x)].update(lazy);
	tree[Rson(x)].update(lazy);
	tree[x].lazy = 0;
}

void build(int x, int l, int r)//建树
{
	tree[x].l = l;
	tree[x].r = r;
	tree[x].color = tree[x].lazy = 0;
	if (l == (r - 1))
	{
		return;
	}
	int mid = (l + r) >> 1;
	build(Lson(x), l, mid);
	build(Rson(x), mid, r);
	push_up(x);
}

void update(int x, int l, int r, int val)
{
	if (tree[x].l >= l && tree[x].r <= r)
	{
		tree[x].update(val);//着色
		return;
	}
	push_down(x);//更新子节点颜色
	if (tree[x].l + 1 == tree[x].r)
		return;
	int mid = (tree[x].l + tree[x].r) >> 1;
	if (l < mid)
		update(Lson(x), l, r, val);
	if (r >= mid)
		update(Rson(x), l, r, val);
	push_up(x);
}

void query(int x, int l, int r)
{
	if (tree[x].color != -1)
	{
		if (vis[tree[x].color] == 0 && tree[x].color != 0)
		{
			vis[tree[x].color] = 1;
		}
		return;
	}
	push_down(x);
	int mid = (tree[x].l + tree[x].r) >> 1;
	query(Lson(x), l, mid);
	query(Rson(x), mid, r);
	push_up(x);
}

int main()
{
	while (~scanf("%d%d", &N, &L))
	{
		ans = 0;
		memset(vis, 0, sizeof(vis));
		for (int i = 0; i < N; i++)
		{
			scanf("%d%d", &l[i], &r[i]);
			num[i << 1] = l[i];
			num[i << 1 | 1] = r[i];
		}
		sort(num, num + (N << 1));
		int size = unique(num, num + (N << 1)) - num;//unique去重,求数组现在的长度
		build(1, 1, size);
		for (int i = 0; i < N; i++)//离散化
		{
			l[i] = (lower_bound(num, num + size, l[i]) - num) + 1;
			r[i] = (lower_bound(num, num + size, r[i]) - num) + 1;
			update(1, l[i], r[i], i + 1);//着色
		}
		query(1, 1, size);
		for (int i = 1; i <= N; i++)
		{
			if (vis[i] != 0)
				ans++;
		}
		printf("%d\n", ans);
	}
	//system("pause");
	return 0;
}
```
